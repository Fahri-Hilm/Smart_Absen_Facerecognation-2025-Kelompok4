```bash
  --name smart-absen \
  -p 5001:5001 \
  --env-file .env \
  -v $(pwd)/logs:/app/logs \
  -v $(pwd)/face_data:/app/face_data \
  --restart unless-stopped \
  ghcr.io/fahri-hilm/smart_absen_facerecognation-2025-kelompok4:latest
```
